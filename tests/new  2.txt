		<script type = "text/JavaScript">
			var GreatestCommonFactor = function(Factor1, Factor2) {
				var GreaterFactor;
				var SmallerFactor;
				if(Factor1 < Factor2) {
					GreaterFactor = Factor2;
					SmallerFactor = Factor1;
				} else {
					GreaterFactor = Factor1;
					SmallerFactor = Factor2;
				}

				var Factor = 1;
				for(var i = 2; i <= SmallerFactor; i++) {
					if(i%GreaterFactor == 0 && i%SmallerFactor == 0) {
						Factor = i;
					}
				}

				return Factor;
				}
		</script>
